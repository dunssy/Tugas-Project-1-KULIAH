#DAN INI ADALAH FUNCTIONNYA UNTUK MENENTUKAN BILANGA
Function BilGenap(input n:integer)->boolean
{menerima input berupa bil n,Menampilkan true jika 
bilangan genap, atau false jika tidak genap}
DEKLARASI :

ALGORITMA : 
  return (n mod 2 = 0 ) 
  


#INI PEMANGGILAN PROGRAM KE FUNCTIONNYA 
PROGRAM PanggilBilanganGenap 
{dinputkan n oleh pengguna, Menampilkan nilai fungsi bilangan}

DEKLARASI 
   n:integer 
  Function BilGenap(input n:integer)->boolean
  
ALGORITMA
  read(n)
  if BilGenap(n) then 
      write("bilangan",n,"adalah genap")
  else
      write("bilangan",n,"adalah bukan genap")
  endif
